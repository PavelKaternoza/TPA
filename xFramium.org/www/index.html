xFramium, a unified testing framework for desktop and mobile web, native, and hybrid apps. Write tests using a simple and elegant XML keyword-driven alternative to coding. Use the same test cases on desktop and mobile web browsers. Take advantage of high-performance parallel execution, advanced caching and intelligent device selection algorithms. Easily connect test data from Excel, XML, JSON, CSV, JDBC and more.

Prefer to code in Java? xFramium also supports Java test code writing using all of the features of xFramium such as the extended page object model.

Test Case 
<?xml version="1.0" encoding="UTF-8"?>
<suite xmlns="http://www.xframium.org/testRegistry" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xsi:schemaLocation="http://www.xframium.org /testRegistry http://www.xframium.org /xsd/pageManagement/testRegistry.xsd">
  <!-- PAGE OBJECTS -->
  <model>
    <page name="Login" />
    <page name="Version" />
  </model>
  <!-- TEST CASES -->
  <test name="VersionTest" dataProvider="versionData" tagNames="smoke">
    <step timed="true" name="CONTEXT" type="DEVICE">
      <parameter type="STATIC" value="NATIVE_APP" />
    </step>
    <step timed="true" name="androidWidget" page="Login" type="CALL" />                 
    <step timed="true" name="USER_NAME" page="Login" type="EXISTS" />
    <step timed="true" name="PASSWORD" page="Login" type="EXISTS" />
    <step timed="true" name="LOGIN_BUTTON" page="Login" type="EXISTS" />
    <step timed="true" name="LOGIN_BUTTON" page="Login" type="CLICK" />                  
  </test>
</suite>
Page Elements defined in many formats to meet any coding style 
<page name="homePage">
	<page name="TITLE" value="Webpage( 'html tag:=h1', 'innerText:=.*Shopping.*')" descriptor="HTML" />
	<page name="CLEAR_CART" value="Link( 'html tag:=a', 'innerText:=.*CLEAR CART.*' )" descriptor="HTML" />
	<page name="CLEAR_ALL" value="Remove Cart Item" descriptor="V_TEXT" />
	<page name="HOME" value="//img[@alt='Logo']" descriptor="XPATH" />
	<page name="DONE" value="/There are no items in your cart" descriptor="V_TEXT" />
	<page name="HOME" value="alt=logo" descriptor="PROP" />
	<page name="HOME" value="resource-id=imageLogo" descriptor="PROP" />
	<page name="PASSCODE_IMAGE" descriptor="V_IMAGE" value="PUBLIC:images/passcodeImage.png" />
</page>
